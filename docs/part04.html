def get_total_participants(survey_id):
    """
    Fetches the total number of participants from the survey.

    Args:
    survey_id (str): The ID of the survey.
 

    Returns:
    int: The total number of participants, or None if an error occurs.
    """
    api_url = f"https://survey.porsline.ir/api/v2/surveys/{survey_id}/responses/result-table/"
    try:
        response = requests.get(api_url, headers=headers)
        response.raise_for_status()  # Raise an exception for bad status codes
 

        data = response.json()  # Parse the JSON response
 

        # --- ADAPT THIS BASED ON THE ACTUAL API RESPONSE STRUCTURE ---
        # Assuming the response contains a 'total_count' or similar field
        # You'll need to inspect the actual response to determine the correct key
        participant_count = data.get('meta', {}).get('total')  # Example: If count is in 'meta'
        
        if participant_count is None:
            participant_count = data.get('total_count')  # Another common key name
        if participant_count is None:
            participant_count = data.get('count') # Try 'count' too
 

        # --- END OF ADAPTATION SECTION ---
 

        if participant_count is not None:
            return int(participant_count)  # Ensure it's an integer
        else:
            return None
 

    except requests.exceptions.RequestException as e:
        print(f"Error: {e}")
        return None
    except json.JSONDecodeError:
        print("Error: Could not parse JSON response.")
        return None
