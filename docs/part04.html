if 'سن شما:' not in df.columns:

else:
    # Define the desired order of age groups
    ordered_ages = [
        'کمتر از ۲۵ سال',
        '۲۶ تا ۳۵ سال',
        '۳۶ تا ۴۵ سال',
        'بالای ۴۵ سال'
    ]

    # Create a mapping from the original age values to the desired order
    age_mapping = {
        age: ordered_age
        for ordered_age in ordered_ages
        for age in df['سن شما:'].unique() if ordered_age in df['سن شما:'].unique() and ordered_age in age
    }

    # Create a categorical column for sorting
    categorical_age = pd.Categorical(df['سن شما:'].map(age_mapping), categories=ordered_ages, ordered=True)
    df_sorted = df.assign(sorted_age=categorical_age).sort_values('sorted_age')

    # Get the unique age groups in the sorted order
    unique_ages_sorted = df_sorted['sorted_age'].unique()

    # Count the occurrences of each age group in the sorted order
    age_counts_sorted = df_sorted['sorted_age'].value_counts().reindex(ordered_ages).fillna(0)
